165Edge<T> * Vertex<T>::addEdge(Vertex<T> *d, double w) {
166 auto newEdge = new Edge<T>(this, d, w);

168 d->incoming.push_back(newEdge);
169 d->setIndegree(d->getIndegree() + 1);
170 return newEdge;

181template <class T>
182bool Vertex<T>::removeEdge(T in) {
183 bool removedEdge = false;
184 auto it = adj.begin();
185 while (it != adj.end()) {
186 Edge<T> *edge = *it;
187 Vertex<T> *dest = edge->getDest();
188 if (dest->getInfo() == in) {
189 it = adj.erase(it);
190 deleteEdge(edge);
191 removedEdge = true; // allows for multiple edges to connect the same pair of vertices (multigraph)
192 }
193 else {
194 it++;
195 }
196 }
197 return removedEdge;
198}

207 auto it = adj.begin();

209 Edge<T> *edge = *it;
210 it = adj.erase(it);
211 deleteEdge(edge);
212 }

391template <class T>
392void Vertex<T>::deleteEdge(Edge<T> *edge) {
393 Vertex<T> *dest = edge->getDest();
394 // Remove the corresponding edge from the incoming list
395 auto it = dest->incoming.begin();
396 while (it != dest->incoming.end()) {
397 if ((*it)->getOrig()->getInfo() == info) {
398 it = dest->incoming.erase(it);
399 }
400 else {
401 it++;
402 }
403 }
404 delete edge;
405}

595template <class T>
596bool Graph<T>::removeVertex(const T &in) {
597 for (auto it = vertexSet.begin(); it != vertexSet.end(); it++) {
598 if ((*it)->getInfo() == in) {
599 auto v = *it;
600 v->removeOutgoingEdges();
601 for (auto u : vertexSet) {
602 u->removeEdge(v->getInfo());
603 }
604 vertexSet.erase(it);
605 delete v;
606 return true;
607 }
608 }
609 return false;
610}

719void Graph<T>::dfsVisit(Vertex<T> *v, std::vector<T> & res) const {
720 v->setVisited(true);
721 res.push_back(v->getInfo());

730/****************** BFS ********************/
738template <class T>
739std::vector<T> Graph<T>::bfs(const T & source) const {
740 std::vector<int> res;
741 // Get the source vertex
742 auto s = findVertex(source);
743 if (s == nullptr) {
744 return res;
745 }
746
747 // Set that no vertex has been visited yet
748 for (auto v : vertexSet) {
749 v->setVisited(false);
750 }
751
752 // Perform the actual BFS using a queue
753 std::queue<Vertex<T> *> q;
754 q.push(s);
755 s->setVisited(true);
756 while (!q.empty()) {
757 auto v = q.front();
758 q.pop();
759 res.push_back(v->getInfo());
760 for (auto & e : v->getAdj()) {
761 auto w = e->getDest();
762 if ( ! w->isVisited()) {
763 q.push(w);
764 w->setVisited(true);
765 }
766 }
767 }
768 return res;
769}

801template <class T>
802bool Graph<T>::dfsIsDAG(Vertex<T> *v) const {
803 v->setVisited(true);
804 v->setProcesssing(true);
805 for (auto e : v->getAdj()) {
806 auto w = e->getDest();
807 if (w->isProcessing()) return false;
808 if (! w->isVisited()) {
809 if (! dfsIsDAG(w)) return false;
810 }
811 }
812 v->setProcesssing(false);
813 return true;
814}
